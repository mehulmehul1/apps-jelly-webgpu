4. **Natural movement** emerges from proper parameter tuning, not complex algorithms

## 6. PARTICULATE.JS INTEGRATION

### Recommended System Architecture

```typescript
/**
 * Particulate.js integration for Abyssal Genesis
 * Based on research from particulatejs.org/docs/
 */
export class JellyfishPhysics {
  // Core particulate systems
  private bellSystem: Particulate.ParticleSystem;
  private tentacleSystems: Particulate.ParticleSystem[] = [];

  // Configuration
  private config: PhysicsConfig;

  // Animation state
  private pulsePhase = 0;
  private time = 0;

  constructor(config: PhysicsConfig) {
    this.config = config;

    // Create bell particle system
    this.bellSystem = this.createBellSystem();

    // Create tentacle systems
    this.tentacleSystems = this.createTentacleSystems();
  }

  /**
   * Create bell (body) particle system
   */
  private createBellSystem(): Particulate.ParticleSystem {
    const system = new Particulate.ParticleSystem();
    const { particles } = this.config;

    // Create spine particles (vertical)
    for (let i = 0; i < particles.bellCount; i++) {
      const y = i * 0.5; // Vertical spacing
      system.addParticle(0, y, 0, 1.0); // x, y, z, mass
    }

    // Create radial rib particles
    const ribCount = 12; // Radial symmetry
    for (let rib = 0; rib < ribCount; rib++) {
      const angle = (rib / ribCount) * Math.PI * 2;
      for (let i = 1; i < particles.bellCount; i++) {
        const r = i * 0.3; // Radius increases with height
        const x = Math.cos(angle) * r;
        const z = Math.sin(angle) * r;
        const y = i * 0.5;
        system.addParticle(x, y, z, 0.5);
      }
    }

    // Add distance constraints for structure
    this.addBellConstraints(system);

    // Pin top particles (anchor points)
    this.addPinConstraints(system);

    return system;
  }

  /**
   * Create tentacle particle systems
   */
  private createTentacleSystems(): Particulate.ParticleSystem[] {
    const systems: Particulate.ParticleSystem[] = [];
    const { tentacles, particles } = this.config;

    for (let group = 0; group < particles.tentacleGroups; group++) {
      const system = new Particulate.ParticleSystem();

      // Create chain particles
      for (let i = 0; i < tentacles.chain.segmentCount; i++) {
        const x = (group - particles.tentacleGroups / 2) * 2;
        const y = -i * tentacles.chain.segmentLength;
        const z = 0;

        // Taper mass toward tip
        const mass = 1.0 - (i / tentacles.chain.segmentCount) * 0.5;
        system.addParticle(x, y, z, mass);
      }

      // Add chain constraints
      for (let i = 1; i < tentacles.chain.segmentCount; i++) {
        system.addConstraint(new Particulate.DistanceConstraint(
          i - 1, i, tentacles.chain.segmentLength
        ));
      }

      // Add angle constraints for bending limits
      if (tentacles.constraints.bendingLimit) {
        for (let i = 1; i < tentacles.chain.segmentCount - 1; i++) {
          system.addConstraint(new Particulate.AngleConstraint(
            i - 1, i, i + 1, tentacles.constraints.bendingLimit
          ));
        }
      }

      systems.push(system);
    }

    return systems;
  }

  /**
   * Update physics simulation
   */
  update(deltaTime: number): void {
    this.time += deltaTime;

    // Update pulse phase
    this.pulsePhase = this.updatePulsePhase(this.time);

    // Apply bell pulsing constraints
    this.applyPulseConstraints(this.pulsePhase);

    // Update flow field forces
    this.applyFlowFieldForces();

    // Step physics
    const iterations = this.config.solver.iterations;
    this.bellSystem.tick(deltaTime, iterations);

    for (const tentacleSystem of this.tentacleSystems) {
      tentacleSystem.tick(deltaTime, iterations);
    }

    // Apply forces
    this.applyForces(deltaTime);
  }

  /**
   * Update pulse phase based on rhythm configuration
   */
  private updatePulsePhase(time: number): number {
    const { movement } = this.config;

    if (movement.type === 'jet-pulse' && movement.jetPulse) {
      const { frequency, variation } = movement.jetPulse;
      const base = Math.sin(time * frequency * Math.PI * 2) * 0.5 + 0.5;

      // Add variation if enabled
      if (variation && variation > 0) {
        const noise = this.perlinNoise(time * 0.5) * variation;
        return Math.max(0, Math.min(1, base + noise));
      }

      return base;
    }

    // Default slow pulse
    return Math.sin(time * 0.5 * Math.PI * 2) * 0.5 + 0.5;
  }

  /**
   * Apply pulse constraints to bell structure
   */
  private applyPulseConstraints(phase: number): void {
    const { movement } = this.config;

    if (movement.type === 'jet-pulse' && movement.jetPulse) {
      const { contraction } = movement.jetPulse;

      // Modify distance constraints based on phase
      // During contraction (phase → 1), reduce rest lengths
      // During recovery (phase → 0), increase rest lengths

      const constraints = this.bellSystem.constraints;
      for (const constraint of constraints) {
        if (constraint instanceof Particulate.DistanceConstraint) {
          const baseLength = constraint.restLength;
          const contractedLength = baseLength * (1 - contraction);
          const currentLength = baseLength * (1 - contraction * phase);
          constraint.restLength = currentLength;
        }
      }
    }
  }

  /**
   * Apply flow field forces to all particles
   */
  private applyFlowFieldForces(): void {
    if (!this.config.flow.enabled) return;

    const { flow } = this.config;
    const positions = this.bellSystem.positions;

    for (let i = 0; i < positions.length; i += 3) {
      const x = positions[i];
      const y = positions[i + 1];
      const z = positions[i + 2];

      // Sample flow field at this position
      const flowForce = this.sampleFlowField(x, y, z, this.time);

      // Apply force to particle velocity
      // (In Verlet, this means modifying position difference)
      positions[i] += flowForce[0] * flow.strength;
      positions[i + 1] += flowForce[1] * flow.strength;
      positions[i + 2] += flowForce[2] * flow.strength;
    }
  }

  /**
   * Sample flow field at position
   */
  private sampleFlowField(x: number, y: number, z: number, time: number): [number, number, number] {
    const { flow } = this.config;
    const { noise, bias } = flow;

    // Generate noise-based flow
    let fx = 0, fy = 0, fz = 0;

    if (noise.type === 'perlin' || noise.type === 'simplex') {
      fx = this.noise3D(x * noise.scale + time * noise.speed, 0, 0) * 2 - 1;
      fy = this.noise3D(0, y * noise.scale + time * noise.speed, 0) * 2 - 1;
      fz = this.noise3D(0, 0, z * noise.scale + time * noise.speed) * 2 - 1;
    }

    // Add directional bias
    fx += bias.x;
    fy += bias.y;
    fz += bias.z;

    return [fx, fy, fz];
  }

  /**
   * Get current particle positions for rendering
   */
  getBellPositions(): Float32Array {
    return this.bellSystem.positions;
  }

  getTentaclePositions(groupIndex: number): Float32Array {
    return this.tentacleSystems[groupIndex].positions;
  }

  // Utility functions
  private perlinNoise(x: number): number {
    // Simplified Perlin noise implementation
    // In production, use a proper noise library
    return Math.sin(x) * 0.5 + 0.5;
  }

  private noise3D(x: number, y: number, z: number): number {
    // Simplified 3D noise
    return Math.sin(x + y + z) * 0.5 + 0.5;
  }

  private addBellConstraints(system: Particulate.ParticleSystem): void {
    // Implementation depends on specific particle layout
    // Add distance constraints for structural integrity
  }

  private addPinConstraints(system: Particulate.ParticleSystem): void {
    // Pin top particles to create anchor points
    system.addConstraint(new Particulate.PinConstraint(0, 0, 10, 0));
  }

  private applyForces(deltaTime: number): void {
    // Apply gravity, buoyancy, drag to all particles
    const { tentacles } = this.config;

    for (const tentacleSystem of this.tentacleSystems) {
      const positions = tentacleSystem.positions;
      const prevPositions = tentacleSystem.positionsPrev;

      for (let i = 0; i < positions.length; i += 3) {
        // Apply gravity
        positions[i + 1] -= tentacles.forces.gravity * deltaTime;

        // Apply buoyancy (opposing gravity)
        positions[i + 1] += tentacles.forces.buoyancy * deltaTime;

        // Apply drag (velocity damping)
        const vx = positions[i] - prevPositions[i];
        const vy = positions[i + 1] - prevPositions[i + 1];
        const vz = positions[i + 2] - prevPositions[i + 2];

        prevPositions[i] = positions[i] - vx * tentacles.forces.drag;
        prevPositions[i + 1] = positions[i + 1] - vy * tentacles.forces.drag;
        prevPositions[i + 2] = positions[i + 2] - vz * tentacles.forces.drag;
      }
    }
  }
}
```

## 8. GROUNDED IN REAL PHYSICS

### Biological Authenticity Guidelines

1. **Reynolds Number Considerations**
   - Small creatures (Re < 1): Viscous forces dominate, movement feels "thick"
   - Large creatures (Re > 1000): Inertial forces dominate, movement feels "coast-y"
   - Most jellyfish: Re 10-1000, transitional regime

2. **Metabolic Constraints**
   - Pulse frequency limited by energy availability
   - Larger creatures pulse slower (allometric scaling)
   - Deep-sea creatures: slower metabolism, reduced activity

3. **Hydrodynamic Efficiency**
   - Jet propulsion: 10-30% efficiency
   - Comb row: 5-15% efficiency
   - Undulation: 20-40% efficiency
   - Energy conservation drives movement patterns

4. **Sensory Response Times**
   - Nerve net transmission: ~1 m/s
   - Reaction delay: 100-500ms
   - Movements are deliberate, not twitchy
